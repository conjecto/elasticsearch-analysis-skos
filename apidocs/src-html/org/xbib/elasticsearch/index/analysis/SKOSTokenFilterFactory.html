002    package org.xbib.elasticsearch.index.analysis;
003    
004    import java.io.IOException;
005    import java.util.LinkedList;
006    import java.util.List;
007    
008    import org.apache.lucene.analysis.TokenStream;
009    import org.apache.lucene.analysis.standard.StandardAnalyzer;
010    import org.apache.lucene.util.Version;
011    
012    import org.elasticsearch.ElasticsearchIllegalArgumentException;
013    import org.elasticsearch.common.inject.Inject;
014    import org.elasticsearch.common.inject.assistedinject.Assisted;
015    import org.elasticsearch.common.settings.Settings;
016    import org.elasticsearch.index.Index;
017    import org.elasticsearch.index.analysis.AbstractTokenFilterFactory;
018    import org.elasticsearch.index.analysis.AnalysisSettingsRequired;
019    import org.elasticsearch.index.settings.IndexSettings;
020    
021    import org.xbib.elasticsearch.index.analysis.skos.SKOSAnalyzer.ExpansionType;
022    import org.xbib.elasticsearch.index.analysis.skos.SKOSLabelFilter;
023    import org.xbib.elasticsearch.index.analysis.skos.SKOSURIFilter;
024    import org.xbib.elasticsearch.index.analysis.skos.engine.SKOSEngine;
025    import org.xbib.elasticsearch.index.analysis.skos.engine.SKOSEngineFactory;
026    import org.xbib.elasticsearch.index.analysis.skos.tokenattributes.SKOSTypeAttribute.SKOSType;
027    
028    @AnalysisSettingsRequired
029    public class SKOSTokenFilterFactory extends AbstractTokenFilterFactory {
030    
031        private final ExpansionType expansionType;
032    
033        private final SKOSEngine skosEngine;
034    
035        private final SKOSType[] type;
036    
037        private final int bufferSize;
038    
039        @Inject
040        public SKOSTokenFilterFactory(Index index, @IndexSettings Settings indexSettings,
041                @Assisted String name, @Assisted Settings settings) {
042            super(index, indexSettings, name, settings);
043            String skosFile = settings.get("skosFile");
044            String expansionTypeString = settings.get("expansionType");
045            String bufferSizeString = settings.get("bufferSize");
046            String languageString = settings.get("language");
047            String typeString = settings.get("skosType");
048            if (skosFile == null) {
049                throw new ElasticsearchIllegalArgumentException("Mandatory parameter 'skosFile' missing");
050            }
051            if (expansionTypeString == null) {
052                throw new ElasticsearchIllegalArgumentException("Mandatory parameter 'expansionType' missing");
053            }
054            if (skosFile.endsWith(".n3") || skosFile.endsWith(".rdf") || skosFile.endsWith(".ttl") || skosFile.endsWith(".zip")) {
055                try {
056                    skosEngine = SKOSEngineFactory.getSKOSEngine(Version.LUCENE_45,
057                            settings.get("path", ""), skosFile,
058                            languageString != null ? languageString.split(" ") : null);
059                } catch (IOException e) {
060                    logger.error(e.getMessage(), e);
061                    throw new ElasticsearchIllegalArgumentException("could not instantiate SKOS engine", e);
062                }
063            } else {
064                throw new ElasticsearchIllegalArgumentException("Allowed file suffixes are: .n3 (N3), .rdf (RDF/XML), .ttl (Turtle) and .zip (zip)");
065            }
066            if (expansionTypeString.equalsIgnoreCase(ExpansionType.LABEL.toString())) {
067                expansionType = ExpansionType.LABEL;
068            } else {
069                expansionType = ExpansionType.URI;
070            }
071            if (bufferSizeString != null) {
072                bufferSize = Integer.parseInt(bufferSizeString);
073                if (bufferSize < 1) {
074                    throw new ElasticsearchIllegalArgumentException("The property 'bufferSize' must be a positive (small) integer");
075                }
076            } else {
077                bufferSize = 4;
078            }
079            List<SKOSType> types = new LinkedList<SKOSType>();
080            if (typeString != null) {
081                for (String s : typeString.split(" ")) {
082                    try {
083                        SKOSType st = SKOSType.valueOf(s.toUpperCase());
084                        if (st != null) {
085                            types.add(st);
086                        }
087                    } catch (IllegalArgumentException e) {
088                        throw new ElasticsearchIllegalArgumentException("The property 'skosType' must be one of PREF, ALT, HIDDEN, BROADER, NARROWER, BROADERTRANSITIVE, NARROWERTRANSITIVE, RELATED");
089                    }
090                }
091            }
092            type = types.toArray(new SKOSType[types.size()]);
093        }
094    
095        @Override
096        public TokenStream create(TokenStream tokenStream) {
097            if (expansionType.equals(ExpansionType.LABEL)) {
098                return new SKOSLabelFilter(tokenStream, skosEngine, new StandardAnalyzer(Version.LUCENE_45), bufferSize, type);
099            } else {
100                return new SKOSURIFilter(tokenStream, skosEngine, new StandardAnalyzer(Version.LUCENE_45), type);
101            }
102        }
103    }
