016    package org.xbib.elasticsearch.index.analysis.skos;
017    
018    import java.io.IOException;
019    import java.io.StringReader;
020    import java.util.Arrays;
021    import java.util.Set;
022    import java.util.Stack;
023    import java.util.TreeSet;
024    
025    import org.apache.lucene.analysis.Analyzer;
026    import org.apache.lucene.analysis.TokenFilter;
027    import org.apache.lucene.analysis.TokenStream;
028    import org.apache.lucene.analysis.payloads.PayloadHelper;
029    import org.apache.lucene.analysis.tokenattributes.CharTermAttribute;
030    import org.apache.lucene.analysis.tokenattributes.PayloadAttribute;
031    import org.apache.lucene.analysis.tokenattributes.PositionIncrementAttribute;
032    import org.apache.lucene.util.AttributeSource;
033    import org.apache.lucene.util.BytesRef;
034    import org.apache.lucene.util.CharsRef;
035    
036    import org.xbib.elasticsearch.index.analysis.skos.engine.SKOSEngine;
037    import org.xbib.elasticsearch.index.analysis.skos.tokenattributes.SKOSTypeAttribute;
038    import org.xbib.elasticsearch.index.analysis.skos.tokenattributes.SKOSTypeAttribute.SKOSType;
039    
040    /**
041     * A SKOS-specific TokenFilter implementation
042     */
043    public abstract class AbstractSKOSFilter extends TokenFilter {
044    
045        /* a stack holding the expanded terms for a token */
046        protected Stack<ExpandedTerm> termStack;
047        /* an engine delivering SKOS concepts */
048        protected SKOSEngine engine;
049        /* the skos types to expand to */
050        protected Set<SKOSType> types;
051        /* provides access to the the term attributes */
052        protected AttributeSource.State current;
053        /* the term text (propagated to the index) */
054        protected final CharTermAttribute termAtt;
055        /* the token position relative to the previous token (propagated) */
056        protected final PositionIncrementAttribute posIncrAtt;
057        /* the binary payload attached to the indexed term (propagated to the index) */
058        protected final PayloadAttribute payloadAtt;
059        /* the SKOS-specific attribute attached to a term */
060        protected final SKOSTypeAttribute skosAtt;
061        /* the analyzer to use when parsing */
062        protected final Analyzer analyzer;
063    
064        /**
065         * Constructor
066         *
067         * @param input the TokenStream
068         * @param engine the engine delivering skos concepts
069         * @param types the skos types to expand to
070         */
071        public AbstractSKOSFilter(TokenStream input, SKOSEngine engine,
072                Analyzer analyzer, SKOSType... types) {
073            super(input);
074            termStack = new Stack<ExpandedTerm>();
075            this.engine = engine;
076            this.analyzer = analyzer;
077    
078            if (types != null && types.length > 0) {
079                this.types = new TreeSet<SKOSType>(Arrays.asList(types));
080            } else {
081                this.types = new TreeSet<SKOSType>(Arrays.asList(new SKOSType[]{
082                            SKOSType.PREF, SKOSType.ALT}));
083            }
084    
085            this.termAtt = addAttribute(CharTermAttribute.class);
086            this.posIncrAtt = addAttribute(PositionIncrementAttribute.class);
087            this.payloadAtt = addAttribute(PayloadAttribute.class);
088            this.skosAtt = addAttribute(SKOSTypeAttribute.class);
089        }
090    
091        /**
092         * Advances the stream to the next token.
093         *
094         * To be implemented by the concrete sub-classes
095         */
096        @Override
097        public abstract boolean incrementToken() throws IOException;
098    
099        /**
100         * Replaces the current term (attributes) with term (attributes) from the
101         * stack
102         *
103         * @throws IOException
104         */
105        protected void processTermOnStack() throws IOException {
106            ExpandedTerm expandedTerm = termStack.pop();
107    
108            String term = expandedTerm.getTerm();
109    
110            SKOSType termType = expandedTerm.getTermType();
111    
112            String sTerm;
113            try {
114                sTerm = analyze(analyzer, term, new CharsRef()).toString();
115            } catch (IllegalArgumentException e) {
116                // skip this term
117                return;
118            }
119    
120            /*
121             * copies the values of all attribute implementations from this state into
122             * the implementations of the target stream
123             */
124            restoreState(current);
125    
126            /*
127             * Adds the expanded term to the term buffer
128             */
129            termAtt.setEmpty().append(sTerm);
130    
131            /*
132             * set position increment to zero to put multiple terms into the same
133             * position
134             */
135            posIncrAtt.setPositionIncrement(0);
136    
137            /*
138             * sets the type of the expanded term (pref, alt, broader, narrower, etc.)
139             */
140            skosAtt.setSkosType(termType);
141    
142            /*
143             * converts the SKOS Attribute to a payload, which is propagated to the
144             * index
145             */
146            byte[] bytes = PayloadHelper.encodeInt(skosAtt.getSkosType().ordinal());
147            payloadAtt.setPayload(new BytesRef(bytes));
148        }
149    
150        /* Snipped from Solr's SynonymMap */
151        public static CharsRef analyze(Analyzer analyzer, String text, CharsRef reuse)
152                throws IOException {
153            TokenStream ts = analyzer.tokenStream("", new StringReader(text));
154            CharTermAttribute termAtt = ts.addAttribute(CharTermAttribute.class);
155            // PositionIncrementAttribute posIncAtt =
156            // ts.addAttribute(PositionIncrementAttribute.class);
157            ts.reset();
158            reuse.length = 0;
159            while (ts.incrementToken()) {
160                int length = termAtt.length();
161                if (length == 0) {
162                    throw new IllegalArgumentException("term: " + text
163                            + " analyzed to a zero-length token");
164                }
165                // if (posIncAtt.getPositionIncrement() != 1) {
166                // throw new IllegalArgumentException("term: " + text +
167                // " analyzed to a token with posinc != 1");
168                // }
169                reuse.grow(reuse.length + length + 1); /* current + word + separator */
170                int end = reuse.offset + reuse.length;
171                if (reuse.length > 0) {
172                    reuse.chars[end++] = 32; // space
173                    reuse.length++;
174                }
175                System.arraycopy(termAtt.buffer(), 0, reuse.chars, end, length);
176                reuse.length += length;
177            }
178            ts.end();
179            ts.close();
180            if (reuse.length == 0) {
181                throw new IllegalArgumentException("term: " + text
182                        + " was completely eliminated by analyzer");
183            }
184            return reuse;
185        }
186    
187        /**
188         * Pushes a given set of labels onto the stack
189         *
190         * @param labels
191         * @param type
192         */
193        protected void pushLabelsToStack(String[] labels, SKOSType type) {
194    
195            if (labels != null) {
196                for (String label : labels) {
197                    termStack.push(new ExpandedTerm(label, type));
198                }
199            }
200    
201        }
202    
203        /**
204         * Helper class for capturing terms and term types
205         */
206        protected static class ExpandedTerm {
207    
208            private final String term;
209            private final SKOSType termType;
210    
211            protected ExpandedTerm(String term, SKOSType termType) {
212                this.term = term;
213                this.termType = termType;
214            }
215    
216            protected String getTerm() {
217                return this.term;
218            }
219    
220            protected SKOSType getTermType() {
221                return this.termType;
222            }
223        }
224    }
