016    package org.xbib.elasticsearch.index.analysis.skos;
017    
018    import java.io.IOException;
019    import java.io.Reader;
020    
021    import org.apache.lucene.analysis.TokenStream;
022    import org.apache.lucene.analysis.core.KeywordTokenizer;
023    import org.apache.lucene.analysis.core.LowerCaseFilter;
024    import org.apache.lucene.analysis.core.StopAnalyzer;
025    import org.apache.lucene.analysis.core.StopFilter;
026    import org.apache.lucene.analysis.miscellaneous.RemoveDuplicatesTokenFilter;
027    import org.apache.lucene.analysis.standard.StandardAnalyzer;
028    import org.apache.lucene.analysis.standard.StandardFilter;
029    import org.apache.lucene.analysis.standard.StandardTokenizer;
030    import org.apache.lucene.analysis.util.CharArraySet;
031    import org.apache.lucene.analysis.util.StopwordAnalyzerBase;
032    import org.apache.lucene.util.Version;
033    
034    import org.xbib.elasticsearch.index.analysis.skos.engine.SKOSEngine;
035    import org.xbib.elasticsearch.index.analysis.skos.engine.SKOSEngineFactory;
036    import org.xbib.elasticsearch.index.analysis.skos.tokenattributes.SKOSTypeAttribute;
037    
038    /**
039     * An analyzer for expanding fields that contain either (i) URI references to
040     * SKOS concepts OR (ii) SKOS concept prefLabels as values.
041     */
042    public class SKOSAnalyzer extends StopwordAnalyzerBase {
043    
044        /**
045         * The supported expansion types
046         */
047        public enum ExpansionType {
048    
049            URI, LABEL
050        }
051        /**
052         * Default expansion type
053         */
054        public static final ExpansionType DEFAULT_EXPANSION_TYPE = ExpansionType.LABEL;
055        private ExpansionType expansionType = DEFAULT_EXPANSION_TYPE;
056        /**
057         * Default skos types to expand to
058         */
059        public static final SKOSTypeAttribute.SKOSType[] DEFAULT_SKOS_TYPES = new SKOSTypeAttribute.SKOSType[]{
060            SKOSTypeAttribute.SKOSType.PREF, SKOSTypeAttribute.SKOSType.ALT, SKOSTypeAttribute.SKOSType.BROADER,
061            SKOSTypeAttribute.SKOSType.BROADERTRANSITIVE, SKOSTypeAttribute.SKOSType.NARROWER,
062            SKOSTypeAttribute.SKOSType.NARROWERTRANSITIVE};
063        private SKOSTypeAttribute.SKOSType[] types = DEFAULT_SKOS_TYPES;
064        /**
065         * A SKOS Engine instance
066         */
067        private SKOSEngine skosEngine;
068        /**
069         * The size of the buffer used for multi-term prediction
070         */
071        private int bufferSize = SKOSLabelFilter.DEFAULT_BUFFER_SIZE;
072        /**
073         * Default maximum allowed token length
074         */
075        public static final int DEFAULT_MAX_TOKEN_LENGTH = 255;
076        private int maxTokenLength = DEFAULT_MAX_TOKEN_LENGTH;
077        /**
078         * An unmodifiable set containing some common English words that are usually
079         * not useful for searching.
080         */
081        public static final CharArraySet STOP_WORDS_SET = StopAnalyzer.ENGLISH_STOP_WORDS_SET;
082    
083        public SKOSAnalyzer(Version matchVersion, CharArraySet stopWords,
084                SKOSEngine skosEngine, ExpansionType expansionType) {
085            super(matchVersion, stopWords);
086            this.skosEngine = skosEngine;
087            this.expansionType = expansionType;
088        }
089    
090        public SKOSAnalyzer(Version matchVersion, SKOSEngine skosEngine,
091                ExpansionType expansionType) {
092            this(matchVersion, STOP_WORDS_SET, skosEngine, expansionType);
093        }
094    
095        public SKOSAnalyzer(Version matchVersion, Reader stopwords,
096                SKOSEngine skosEngine, ExpansionType expansionType) throws IOException {
097            this(matchVersion, loadStopwordSet(stopwords, matchVersion), skosEngine,
098                    expansionType);
099        }
100    
101        public SKOSAnalyzer(Version matchVersion, CharArraySet stopWords,
102                String indexPath, String skosFile,
103                ExpansionType expansionType, int bufferSize, String... languages)
104                throws IOException {
105            super(matchVersion, stopWords);
106            this.skosEngine = SKOSEngineFactory.getSKOSEngine(matchVersion, indexPath, skosFile,
107                    languages);
108            this.expansionType = expansionType;
109            this.bufferSize = bufferSize;
110        }
111    
112        public SKOSAnalyzer(Version matchVersion, 
113                String indexPath, String skosFile,
114                ExpansionType expansionType, int bufferSize, String... languages)
115                throws IOException {
116            this(matchVersion, STOP_WORDS_SET, indexPath,  skosFile, expansionType, bufferSize,
117                    languages);
118        }
119    
120        public SKOSAnalyzer(Version matchVersion, 
121                String indexPath, String skosFile,
122                ExpansionType expansionType, int bufferSize) throws IOException {
123            this(matchVersion, indexPath, skosFile, expansionType, bufferSize, (String[]) null);
124        }
125    
126        public SKOSAnalyzer(Version matchVersion, 
127                String indexPath, String skosFile,
128                ExpansionType expansionType) throws IOException {
129            this(matchVersion, indexPath, skosFile, expansionType,
130                    SKOSLabelFilter.DEFAULT_BUFFER_SIZE);
131        }
132    
133        public SKOSAnalyzer(Version matchVersion, Reader stopwords, 
134                String indexPath, String skosFile,
135                ExpansionType expansionType, int bufferSize, String... languages)
136                throws IOException {
137            this(matchVersion, loadStopwordSet(stopwords, matchVersion), indexPath, skosFile,
138                    expansionType, bufferSize, languages);
139        }
140    
141        public SKOSTypeAttribute.SKOSType[] getTypes() {
142            return types;
143        }
144    
145        public void setTypes(SKOSTypeAttribute.SKOSType... types) {
146            this.types = types;
147        }
148    
149        /**
150         * Set maximum allowed token length. If a token is seen that exceeds this
151         * length then it is discarded. This setting only takes effect the next time
152         * tokenStream or tokenStream is called.
153         */
154        public void setMaxTokenLength(int length) {
155            maxTokenLength = length;
156        }
157    
158        /**
159         * @see #setMaxTokenLength
160         */
161        public int getMaxTokenLength() {
162            return maxTokenLength;
163        }
164    
165        @Override
166        protected TokenStreamComponents createComponents(String fileName,
167                Reader reader) {
168            if (expansionType.equals(ExpansionType.URI)) {
169                final KeywordTokenizer src = new KeywordTokenizer(reader);
170                TokenStream tok = new SKOSURIFilter(src, skosEngine,
171                        new StandardAnalyzer(matchVersion), types);
172                tok = new LowerCaseFilter(matchVersion, tok);
173                return new TokenStreamComponents(src, tok);
174            } else {
175                final StandardTokenizer src = new StandardTokenizer(matchVersion, reader);
176                src.setMaxTokenLength(maxTokenLength);
177                TokenStream tok = new StandardFilter(matchVersion, src);
178                // prior to this we get the classic behavior, standardfilter does it for
179                // us.
180                tok = new SKOSLabelFilter(tok, skosEngine, new StandardAnalyzer(
181                        matchVersion), bufferSize, types);
182                tok = new LowerCaseFilter(matchVersion, tok);
183                tok = new StopFilter(matchVersion, tok, stopwords);
184                tok = new RemoveDuplicatesTokenFilter(tok);
185                return new TokenStreamComponents(src, tok) {
186                    @Override
187                    protected void setReader(final Reader reader) throws IOException {
188                        src.setMaxTokenLength(maxTokenLength);
189                        super.setReader(reader);
190                    }
191                };
192            }
193        }
194    }
