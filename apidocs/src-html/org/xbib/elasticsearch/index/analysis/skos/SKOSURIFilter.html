016    package org.xbib.elasticsearch.index.analysis.skos;
017    
018    import java.io.IOException;
019    
020    import org.apache.lucene.analysis.Analyzer;
021    import org.apache.lucene.analysis.TokenStream;
022    
023    import org.xbib.elasticsearch.index.analysis.skos.engine.SKOSEngine;
024    import org.xbib.elasticsearch.index.analysis.skos.tokenattributes.SKOSTypeAttribute.SKOSType;
025    
026    /**
027     * A Lucene TokenFilter that supports URI-based term expansion as described in
028     * https://code.
029     * google.com/p/lucene-skos/wiki/UseCases#UC1:_URI-based_term_expansion
030     *
031     * It takes references to SKOS concepts (URIs) as input and searches a given
032     * SKOS vocabulary for matching concepts. If a match is found, it adds the
033     * concept's labels to the output token stream.
034     */
035    public final class SKOSURIFilter extends AbstractSKOSFilter {
036    
037        /**
038         * Constructor.
039         *
040         * @param input
041         * @param skosEngine
042         * @param types
043         */
044        public SKOSURIFilter(TokenStream input, SKOSEngine skosEngine,
045                Analyzer analyzer, SKOSType... types) {
046            super(input, skosEngine, analyzer, types);
047        }
048    
049        /**
050         * Advances the stream to the next token
051         */
052        @Override
053        public boolean incrementToken() throws IOException {
054    
055            /* there are expanded terms for the given token */
056            if (termStack.size() > 0) {
057                processTermOnStack();
058                return true;
059            }
060    
061            /* no more tokens on the consumed stream -> end of stream */
062            if (!input.incrementToken()) {
063                return false;
064            }
065    
066            /* check whether there are expanded terms for a given token */
067            if (addTermsToStack(termAtt.toString())) {
068    
069                /* if yes, capture the state of all attributes */
070                current = captureState();
071            }
072    
073            return true;
074        }
075    
076        /**
077         * Assumes that the given term is a concept URI
078         */
079        public boolean addTermsToStack(String term) throws IOException {
080            try {
081                if (types.contains(SKOSType.PREF)) {
082                    String[] prefLabels = engine.getPrefLabels(term);
083                    pushLabelsToStack(prefLabels, SKOSType.PREF);
084                }
085                if (types.contains(SKOSType.ALT)) {
086                    String[] altLabels = engine.getAltLabels(term);
087                    pushLabelsToStack(altLabels, SKOSType.ALT);
088                }
089                if (types.contains(SKOSType.BROADER)) {
090                    String[] broaderLabels = engine.getBroaderLabels(term);
091                    pushLabelsToStack(broaderLabels, SKOSType.BROADER);
092                }
093                if (types.contains(SKOSType.BROADERTRANSITIVE)) {
094                    String[] broaderTransitiveLabels = engine
095                            .getBroaderTransitiveLabels(term);
096                    pushLabelsToStack(broaderTransitiveLabels, SKOSType.BROADERTRANSITIVE);
097                }
098                if (types.contains(SKOSType.NARROWER)) {
099                    String[] narrowerLabels = engine.getNarrowerLabels(term);
100                    pushLabelsToStack(narrowerLabels, SKOSType.NARROWER);
101                }
102                if (types.contains(SKOSType.NARROWERTRANSITIVE)) {
103                    String[] narrowerTransitiveLabels = engine
104                            .getNarrowerTransitiveLabels(term);
105                    pushLabelsToStack(narrowerTransitiveLabels, SKOSType.NARROWERTRANSITIVE);
106                }
107            } catch (Exception e) {
108                System.err
109                        .println("Error when accessing SKOS Engine.\n" + e.getMessage());
110            }
111    
112            if (termStack.isEmpty()) {
113                return false;
114            }
115    
116            return true;
117        }
118    }
