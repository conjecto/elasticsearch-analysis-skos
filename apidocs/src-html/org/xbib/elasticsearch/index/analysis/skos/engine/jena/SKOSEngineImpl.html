150        /**
151         * This constructor loads the SKOS model from a given InputStream using the
152         * given serialization language parameter, which must be either N3, RDF/XML,
153         * or TURTLE.
154         *
155         * @param inputStream the input stream
156         * @param lang the serialization language
157         * @throws IOException if the model cannot be loaded
158         */
159        public SKOSEngineImpl(final Version version, InputStream inputStream,
160                String lang) throws IOException {
161    
162            if (!("N3".equals(lang) || "RDF/XML".equals(lang) || "TURTLE".equals(lang))) {
163                throw new IOException("Invalid RDF serialization format");
164            }
165    
166            matchVersion = version;
167    
168            analyzer = new SimpleAnalyzer(matchVersion);
169    
170            skosModel = ModelFactory.createDefaultModel();
171    
172            skosModel.read(inputStream, null, lang);
173    
174            indexDir = new RAMDirectory();
175            entailSKOSModel();
176            indexSKOSModel();
177    
178            searcher = new IndexSearcher(DirectoryReader.open(indexDir));
179        }

192        /**
193         * This constructor loads the SKOS model from a given filename or URI,
194         * starts the indexing process and sets up the index searcher.
195         *
196         * @param languages the languages to be considered
197         * @param indexPath index path
198         * @param filenameOrURI file name or URI
199         * @throws IOException
200         */
201        public SKOSEngineImpl(final Version version, String indexPath, String filenameOrURI,
202                String... languages) throws IOException {
203            matchVersion = version;
204            analyzer = new SimpleAnalyzer(matchVersion);
205    
206            String langSig = "";
207            if (languages != null) {
208                this.languages = new TreeSet<String>(Arrays.asList(languages));
209                langSig = "-" + join(this.languages.iterator(), '.');
210            }
211    
212            String name = getName(filenameOrURI);
213            File dir = new File(indexPath + name + langSig);
214    
215            indexDir = FSDirectory.open(dir);
216            if (!dir.isDirectory()) {
217    
218                // load the skos model from the given file
219                FileManager fileManager = new FileManager();
220                fileManager.addLocatorFile();
221                fileManager.addLocatorURL();
222                fileManager.addLocatorClassLoader(SKOSEngineImpl.class.getClassLoader());
223    
224                if (getExtension(filenameOrURI).equals("zip")) {
225                    fileManager.addLocatorZip(filenameOrURI);
226                    filenameOrURI = getBaseName(filenameOrURI);
227                }
228    
229                skosModel = fileManager.loadModel(filenameOrURI);
230                entailSKOSModel();
231                indexSKOSModel();
232            }
233    
234            searcher = new IndexSearcher(DirectoryReader.open(indexDir));
235        }

237        private void entailSKOSModel() {
238            GraphStore graphStore = GraphStoreFactory.create(skosModel);
239            String sparqlQuery =
240                    "PREFIX skos: <http://www.w3.org/2004/02/skos/core#>\n"
241                    + "PREFIX rdf:<http://www.w3.org/1999/02/22-rdf-syntax-ns#>\n"
242                    + "INSERT { ?subject rdf:type skos:Concept }\n"
243                    + "WHERE {\n"
244                    + "{ ?subject skos:prefLabel ?text } UNION\n"
245                    + "{ ?subject skos:altLabel ?text } UNION\n"
246                    + "{ ?subject skos:hiddenLabel ?text }\n"
247                    + "}";
248            UpdateRequest request = UpdateFactory.create(sparqlQuery);
249            UpdateAction.execute(request, graphStore);
250        }

252        /**
253         * Creates lucene documents from SKOS concept. In order to allow language
254         * restrictions, one document per language is created.
255         */
256        private Document createDocumentsFromConcept(Resource skos_concept) {
257            Document conceptDoc = new Document();
258    
259            String conceptURI = skos_concept.getURI();
260            Field uriField = new Field(FIELD_URI, conceptURI, StringField.TYPE_STORED);
261            conceptDoc.add(uriField);
262    
263            // store the preferred lexical labels
264            indexAnnotation(skos_concept, conceptDoc, SKOS.prefLabel, FIELD_PREF_LABEL);
265    
266            // store the alternative lexical labels
267            indexAnnotation(skos_concept, conceptDoc, SKOS.altLabel, FIELD_ALT_LABEL);
268    
269            // store the hidden lexical labels
270            indexAnnotation(skos_concept, conceptDoc, SKOS.hiddenLabel,
271                    FIELD_HIDDEN_LABEL);
272    
273            // store the URIs of the broader concepts
274            indexObject(skos_concept, conceptDoc, SKOS.broader, FIELD_BROADER);
275    
276            // store the URIs of the broader transitive concepts
277            indexObject(skos_concept, conceptDoc, SKOS.broaderTransitive,
278                    FIELD_BROADER_TRANSITIVE);
279    
280            // store the URIs of the narrower concepts
281            indexObject(skos_concept, conceptDoc, SKOS.narrower, FIELD_NARROWER);
282    
283            // store the URIs of the narrower transitive concepts
284            indexObject(skos_concept, conceptDoc, SKOS.narrowerTransitive,
285                    FIELD_NARROWER_TRANSITIVE);
286    
287            // store the URIs of the related concepts
288            indexObject(skos_concept, conceptDoc, SKOS.related, FIELD_RELATED);
289    
290            return conceptDoc;
291        }

298        @Override
299        public String[] getAltTerms(String label) throws IOException {
300            List<String> result = new ArrayList<String>();
301    
302            // convert the query to lower-case
303            String queryString = label.toLowerCase();
304    
305            try {
306                String[] conceptURIs = getConcepts(queryString);
307    
308                for (String conceptURI : conceptURIs) {
309                    String[] altLabels = getAltLabels(conceptURI);
310                    if (altLabels != null) {
311                        result.addAll(Arrays.asList(altLabels));
312                    }
313                }
314            } catch (Exception e) {
315                System.err
316                        .println("Error when accessing SKOS Engine.\n" + e.getMessage());
317            }
318    
319            return result.toArray(new String[result.size()]);
320        }

349        @Override
350        public String[] getConcepts(String label) throws IOException {
351            List<String> concepts = new ArrayList<String>();
352    
353            // convert the query to lower-case
354            String queryString = label.toLowerCase();
355    
356            AllDocCollector collector = new AllDocCollector();
357    
358            DisjunctionMaxQuery query = new DisjunctionMaxQuery(0.0f);
359            query.add(new TermQuery(new Term(FIELD_PREF_LABEL, queryString)));
360            query.add(new TermQuery(new Term(FIELD_ALT_LABEL, queryString)));
361            query.add(new TermQuery(new Term(FIELD_HIDDEN_LABEL, queryString)));
362            searcher.search(query, collector);
363    
364            for (Integer hit : collector.getDocs()) {
365                Document doc = searcher.doc(hit);
366                String conceptURI = doc.getValues(FIELD_URI)[0];
367                concepts.add(conceptURI);
368            }
369    
370            return concepts.toArray(new String[concepts.size()]);
371        }

373        private String[] getLabels(String conceptURI, String field)
374                throws IOException {
375            List<String> labels = new ArrayList<String>();
376            String[] concepts = readConceptFieldValues(conceptURI, field);
377    
378            for (String aConceptURI : concepts) {
379                String[] prefLabels = getPrefLabels(aConceptURI);
380                labels.addAll(Arrays.asList(prefLabels));
381    
382                String[] altLabels = getAltLabels(aConceptURI);
383                labels.addAll(Arrays.asList(altLabels));
384            }
385    
386            return labels.toArray(new String[labels.size()]);
387        }

426        private void indexAnnotation(Resource skos_concept, Document conceptDoc,
427                AnnotationProperty property, String field) {
428            StmtIterator stmt_iter = skos_concept.listProperties(property);
429            while (stmt_iter.hasNext()) {
430                Literal labelLiteral = stmt_iter.nextStatement().getObject()
431                        .as(Literal.class);
432                String label = labelLiteral.getLexicalForm();
433                String labelLang = labelLiteral.getLanguage();
434    
435                if (this.languages != null && !this.languages.contains(labelLang)) {
436                    continue;
437                }
438    
439                // converting label to lower-case
440                label = label.toLowerCase();
441    
442                Field labelField = new Field(field, label, StringField.TYPE_STORED);
443    
444                conceptDoc.add(labelField);
445            }
446        }
447    
448        private void indexObject(Resource skos_concept, Document conceptDoc,
449                ObjectProperty property, String field) {
450            StmtIterator stmt_iter = skos_concept.listProperties(property);
451            while (stmt_iter.hasNext()) {
452                RDFNode concept = stmt_iter.nextStatement().getObject();
453    
454                if (!concept.canAs(Resource.class)) {
455                    System.err.println("Error when indexing relationship of concept "
456                            + skos_concept.getURI() + " .");
457                    continue;
458                }
459    
460                Resource resource = concept.as(Resource.class);
461    
462                Field conceptField = new Field(field, resource.getURI(),
463                        TextField.TYPE_STORED);
464    
465                conceptDoc.add(conceptField);
466            }
467        }

469        /**
470         * Creates the synonym index
471         *
472         * @throws IOException
473         */
474        private void indexSKOSModel() throws IOException {
475            IndexWriterConfig cfg = new IndexWriterConfig(matchVersion, analyzer);
476            IndexWriter writer = new IndexWriter(indexDir, cfg);
477            writer.getConfig().setRAMBufferSizeMB(48);
478    
479            /* iterate SKOS concepts, create Lucene docs and add them to the index */
480            ResIterator concept_iter = skosModel.listResourcesWithProperty(RDF.type,
481                    SKOS.Concept);
482            while (concept_iter.hasNext()) {
483                Resource skos_concept = concept_iter.next();
484    
485                Document concept_doc = createDocumentsFromConcept(skos_concept);
486    
487                writer.addDocument(concept_doc);
488            }
489    
490            writer.close();
491        }

493        /**
494         * Returns the values of a given field for a given concept
495         */
496        private String[] readConceptFieldValues(String conceptURI, String field)
497                throws IOException {
498    
499            Query query = new TermQuery(new Term(FIELD_URI, conceptURI));
500    
501            TopDocs docs = searcher.search(query, 1);
502    
503            ScoreDoc[] results = docs.scoreDocs;
504    
505            if (results.length != 1) {
506                System.out.println("Unknown concept " + conceptURI);
507                return null;
508            }
509    
510            Document conceptDoc = searcher.doc(results[0].doc);
511    
512            return conceptDoc.getValues(field);
513        }

515        private String join(Iterator iterator, char separator) {
516            // handle null, zero and one elements before building a buffer
517            if (iterator == null) {
518                return null;
519            }
520            if (!iterator.hasNext()) {
521                return "";
522            }
523            Object first = iterator.next();
524            if (!iterator.hasNext()) {
525                return first == null ? "" : first.toString();
526            }
527            // two or more elements
528            StringBuilder buf = new StringBuilder();
529            if (first != null) {
530                buf.append(first);
531            }
532            while (iterator.hasNext()) {
533                buf.append(separator);
534                Object obj = iterator.next();
535                if (obj != null) {
536                    buf.append(obj);
537                }
538            }
539            return buf.toString();
540        }

542        private String getName(String filename) {
543            if (filename == null) {
544                return null;
545            }
546            int index = indexOfLastSeparator(filename);
547            return filename.substring(index + 1);
548        }
549        private static final char UNIX_SEPARATOR = '/';
550        private static final char WINDOWS_SEPARATOR = '\\';
551    
552        private int indexOfLastSeparator(String filename) {
553            if (filename == null) {
554                return -1;
555            }
556            int lastUnixPos = filename.lastIndexOf(UNIX_SEPARATOR);
557            int lastWindowsPos = filename.lastIndexOf(WINDOWS_SEPARATOR);
558            return Math.max(lastUnixPos, lastWindowsPos);
559        }
560    
561        private String getExtension(String filename) {
562            if (filename == null) {
563                return null;
564            }
565            int index = indexOfExtension(filename);
566            if (index == -1) {
567                return "";
568            } else {
569                return filename.substring(index + 1);
570            }
571        }
572    
573        private int indexOfExtension(String filename) {
574            if (filename == null) {
575                return -1;
576            }
577            int extensionPos = filename.lastIndexOf(EXTENSION_SEPARATOR);
578            int lastSeparator = indexOfLastSeparator(filename);
579            return (lastSeparator > extensionPos ? -1 : extensionPos);
580        }
581        public static final char EXTENSION_SEPARATOR = '.';
582    
583        private String getBaseName(String filename) {
584            return removeExtension(getName(filename));
585        }
586    
587        private String removeExtension(String filename) {
588            if (filename == null) {
589                return null;
590            }
591            int index = indexOfExtension(filename);
592            if (index == -1) {
593                return filename;
594            } else {
595                return filename.substring(0, index);
596            }
597        }
