016    package org.xbib.elasticsearch.index.analysis.skos.tokenattributes;
017    
018    import org.apache.lucene.util.Attribute;
019    
020    /**
021     * This class represents SKOS-specific meta-information that is assigned to
022     * tokens during the analysis phase.
023     *
024     * Note: when tokens are posted to the index as terms, attribute information is
025     * lost unless it is encoded in the terms' payload.
026     */
027    public interface SKOSTypeAttribute extends Attribute {
028    
029        /**
030         * An enumeration of supported SKOS concept types
031         */
032        public static enum SKOSType {
033    
034            PREF, ALT, HIDDEN, BROADER, NARROWER, BROADERTRANSITIVE, NARROWERTRANSITIVE, RELATED;
035    
036            /**
037             * Returns the SKOSType given the ordinal.
038             */
039            public static SKOSType fromInteger(int ordinal) {
040                switch (ordinal) {
041                    case 0:
042                        return PREF;
043                    case 1:
044                        return ALT;
045                    case 2:
046                        return HIDDEN;
047                    case 3:
048                        return BROADER;
049                    case 4:
050                        return NARROWER;
051                    case 5:
052                        return BROADERTRANSITIVE;
053                    case 6:
054                        return NARROWERTRANSITIVE;
055                    case 7:
056                        return RELATED;
057                    default:
058                        return RELATED;
059                }
060            }
061        }
062    
063        /**
064         * Returns the SKOS type
065         *
066         * @return SKOSType
067         */
068        SKOSType getSkosType();
069    
070        /**
071         * Sets this Token's SKOSType.
072         *
073         * @param skosType
074         */
075        void setSkosType(SKOSType skosType);
076    }
