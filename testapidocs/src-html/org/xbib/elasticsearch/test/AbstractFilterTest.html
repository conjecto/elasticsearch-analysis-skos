031    /**
032     * Abstract class containing common code for filter tests
033     */

036        protected final Version matchVersion = Version.LUCENE_45;
037    
038        protected IndexSearcher searcher;
039    
040        protected IndexWriter writer;
041    
042        protected SKOSEngineMock skosEngine;
043    
044        protected SKOSAnalyzer skosAnalyzer;
045    
046        protected Directory directory;
047    
048        @BeforeTest
049        protected void setUp() throws Exception {
050    
051            // adding some test data
052            skosEngine = new SKOSEngineMock();
053    
054            skosEngine.addEntry("http://example.com/concept/1", SKOSType.PREF, "jumps");
055            skosEngine.addEntry("http://example.com/concept/1", SKOSType.ALT, "leaps",
056                    "hops");
057    
058            skosEngine.addEntry("http://example.com/concept/2", SKOSType.PREF, "quick");
059            skosEngine.addEntry("http://example.com/concept/2", SKOSType.ALT, "fast",
060                    "speedy");
061    
062            skosEngine.addEntry("http://example.com/concept/3", SKOSType.PREF, "over");
063            skosEngine.addEntry("http://example.com/concept/3", SKOSType.ALT, "above");
064    
065            skosEngine.addEntry("http://example.com/concept/4", SKOSType.PREF, "lazy");
066            skosEngine.addEntry("http://example.com/concept/4", SKOSType.ALT,
067                    "apathic", "sluggish");
068    
069            skosEngine.addEntry("http://example.com/concept/5", SKOSType.PREF, "dog");
070            skosEngine.addEntry("http://example.com/concept/5", SKOSType.ALT, "canine",
071                    "pooch");
072    
073            skosEngine.addEntry("http://example.com/concept/6", SKOSType.PREF,
074                    "united nations");
075            skosEngine.addEntry("http://example.com/concept/6", SKOSType.ALT, "UN");
076    
077            skosEngine.addEntry("http://example.com/concept/7", SKOSType.PREF,
078                    "lazy dog");
079            skosEngine.addEntry("http://example.com/concept/7", SKOSType.ALT, "Odie");
080    
081            directory = new RAMDirectory();
082    
083        }
084    
085        @AfterTest
086        public void tearDown() throws Exception {
087    
088            if (writer != null) {
089                writer.close();
090            }
091    
092            if (searcher != null) {
093                searcher.getIndexReader().close();
094            }
095    
096        }
097    }
