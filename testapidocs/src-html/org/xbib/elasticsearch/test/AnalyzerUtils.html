016    package org.xbib.elasticsearch.test;
017    
018    import java.io.IOException;
019    import java.io.StringReader;
020    
021    import org.apache.lucene.analysis.Analyzer;
022    import org.apache.lucene.analysis.TokenStream;
023    import org.apache.lucene.analysis.tokenattributes.CharTermAttribute;
024    import org.apache.lucene.analysis.tokenattributes.OffsetAttribute;
025    import org.apache.lucene.analysis.tokenattributes.PayloadAttribute;
026    import org.apache.lucene.analysis.tokenattributes.PositionIncrementAttribute;
027    import org.apache.lucene.analysis.tokenattributes.TypeAttribute;
028    import org.apache.lucene.util.BytesRef;
029    
030    /**
031     * Utils for displaying the results of the Lucene analysis process
032     */
033    public class AnalyzerUtils {
034    
035        //private final static Logger logger = Logg
036    
037        public static void displayTokens(Analyzer analyzer, String text)
038                throws IOException {
039            displayTokens(analyzer.tokenStream("contents", new StringReader(text)));
040        }
041    
042        public static void displayTokens(TokenStream stream) throws IOException {
043    
044            CharTermAttribute term = stream.addAttribute(CharTermAttribute.class);
045            while (stream.incrementToken()) {
046                System.out.println("[" + term.toString() + "] ");
047            }
048    
049        }
050    
051        public static void displayTokensWithPositions(Analyzer analyzer, String text)
052                throws IOException {
053    
054            TokenStream stream = analyzer.tokenStream("contents",
055                    new StringReader(text));
056    
057            CharTermAttribute term = stream.addAttribute(CharTermAttribute.class);
058            PositionIncrementAttribute posIncr = stream
059                    .addAttribute(PositionIncrementAttribute.class);
060    
061            int position = 0;
062            while (stream.incrementToken()) {
063    
064                int increment = posIncr.getPositionIncrement();
065                if (increment > 0) {
066                    position = position + increment;
067                    System.out.println();
068                    System.out.print(position + ":");
069                }
070    
071                System.out.print("[" + term.toString() + "] ");
072    
073            }
074            System.out.println();
075    
076        }
077    
078        public static void displayTokensWithFullDetails(Analyzer analyzer, String text)
079                throws IOException {
080    
081            TokenStream stream = analyzer.tokenStream("contents",
082                    new StringReader(text));
083    
084            CharTermAttribute term = stream.addAttribute(CharTermAttribute.class);
085            PositionIncrementAttribute posIncr = stream
086                    .addAttribute(PositionIncrementAttribute.class);
087            OffsetAttribute offset = stream.addAttribute(OffsetAttribute.class);
088            TypeAttribute type = stream.addAttribute(TypeAttribute.class);
089            PayloadAttribute payload = stream.addAttribute(PayloadAttribute.class);
090    
091            int position = 0;
092            while (stream.incrementToken()) {
093    
094                int increment = posIncr.getPositionIncrement();
095                if (increment > 0) {
096                    position = position + increment;
097                    System.out.println();
098                    System.out.print(position + ":");
099                }
100    
101                BytesRef pl = payload.getPayload();
102    
103                if (pl != null) {
104                    System.out.print("[" + term.toString() + ":" + offset.startOffset()
105                            + "->" + offset.endOffset() + ":" + type.type() + ":"
106                            + new String(pl.bytes) + "] ");
107    
108                } else {
109                    System.out.print("[" + term.toString() + ":" + offset.startOffset()
110                            + "->" + offset.endOffset() + ":" + type.type() + "] ");
111    
112                }
113            }
114            System.out.println();
115        }
116    }
