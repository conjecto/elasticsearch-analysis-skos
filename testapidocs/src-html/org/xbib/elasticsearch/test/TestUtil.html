016    package org.xbib.elasticsearch.test;
017    
018    import java.io.IOException;
019    import org.apache.lucene.search.Explanation;
020    import org.apache.lucene.search.IndexSearcher;
021    import org.apache.lucene.search.Query;
022    import org.apache.lucene.search.ScoreDoc;
023    import org.apache.lucene.search.TopDocs;
024    
025    /**
026     * Common utilities for testing.
027     */
028    public class TestUtil {
029    
030        public static int hitCount(IndexSearcher searcher, Query query)
031                throws IOException {
032            TopDocs results = searcher.search(query, 1);
033            return results.totalHits;
034        }
035    
036        public static void explainQuery(IndexSearcher searcher, Query query)
037                throws IOException {
038            TopDocs topDocs = searcher.search(query, 10);
039            for (ScoreDoc match : topDocs.scoreDocs) {
040                Explanation explanation = searcher.explain(query, match.doc);
041                System.out.println("---------------");
042                System.out.println(explanation.toString());
043            }
044        }
045    }
