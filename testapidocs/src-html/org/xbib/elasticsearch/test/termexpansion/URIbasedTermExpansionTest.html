046    /**
047     * This test-case verifies and demonstrates the "Expansion of URI terms to SKOS
048     * labels" use case as described in https://code.
049     * google.com/p/lucene-skos/wiki/UseCases#UC1:_URI-based_term_expansion
050     */
051    public class URIbasedTermExpansionTest extends AbstractTermExpansionTest {
052    
053        /**
054         * This test indexes a sample metadata record (=lucene document) having a
055         * "title", "description", and "subject" field, which is semantically
056         * enriched by a URI pointing to a SKOS concept "weapons".
057         *
058         * A search for "arms" returns that record as a result because "arms" is
059         * defined as an alternative label (altLabel) for the concept "weapons".
060         *
061         * @throws IOException
062         */
063        @Test
064        public void uriBasedTermExpansion() throws IOException {
065    
066            /* defining the document to be indexed */
067            Document doc = new Document();
068            doc.add(new Field("title", "Spearhead",
069                    TextField.TYPE_STORED));
070            doc.add(new Field(
071                    "description",
072                    "Roman iron spearhead. The spearhead was attached to one end of a wooden shaft..."
073                    + "The spear was mainly a thrusting weapon, but could also be thrown. "
074                    + "It was the principal weapon of the auxiliary soldier... "
075                    + "(second - fourth century, Arbeia Roman Fort).",
076                    TextField.TYPE_NOT_STORED));
077            doc.add(new Field("subject",
078                    "http://www.ukat.org.uk/thesaurus/concept/859",
079                    TextField.TYPE_NOT_STORED));
080    
081            /* setting up the SKOS analyzer */
082            String skosFile = "src/test/resources/skos_samples/ukat_examples.n3";
083            String indexPath = "target/";
084    
085            /* ExpansionType.URI->the field to be analyzed (expanded) contains URIs */
086            Analyzer skosAnalyzer = new SKOSAnalyzer(matchVersion, indexPath, skosFile,
087                    ExpansionType.URI);
088    
089            /* Define different analyzers for different fields */
090            Map<String, Analyzer> analyzerPerField = new HashMap<String, Analyzer>();
091            analyzerPerField.put("subject", skosAnalyzer);
092            PerFieldAnalyzerWrapper indexAnalyzer = new PerFieldAnalyzerWrapper(
093                    new SimpleAnalyzer(matchVersion), analyzerPerField);
094    
095            /* setting up a writer with a default (simple) analyzer */
096            writer = new IndexWriter(new RAMDirectory(), new IndexWriterConfig(
097                    matchVersion, indexAnalyzer));
098    
099            /* adding the document to the index */
100            writer.addDocument(doc);
101    
102            /* defining a query that searches over all fields */
103            BooleanQuery query1 = new BooleanQuery();
104            query1.add(new TermQuery(new Term("title", "arms")),
105                    BooleanClause.Occur.SHOULD);
106            query1.add(new TermQuery(new Term("description", "arms")),
107                    BooleanClause.Occur.SHOULD);
108            query1.add(new TermQuery(new Term("subject", "arms")),
109                    BooleanClause.Occur.SHOULD);
110    
111            /* creating a new searcher */
112            searcher = new IndexSearcher(DirectoryReader.open(writer, false));
113    
114            TopDocs results = searcher.search(query1, 10);
115    
116            /* the document matches because "arms" is among the expanded terms */
117            Assert.assertEquals(1, results.totalHits);
118    
119            /* defining a query that searches for a broader concept */
120            Query query2 = new TermQuery(new Term("subject", "military equipment"));
121    
122            results = searcher.search(query2, 10);
123    
124            /* ... also returns the document as result */
125            Assert.assertEquals(1, results.totalHits);
126    
127        }
128    }
